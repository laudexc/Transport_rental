# Структура проекта

```text
rental-system/
├── cmd/
│   └── main.go                     # точка входа (CLI / тестовый запуск)
│
├── internal/
│   ├── transport/                  # всё, что связано с транспортом
│   │   ├── interfaces/
│   │   │   ├── vehicle.go          # базовый интерфейс транспорта
│   │   │   ├── rentable.go         # всё, что можно арендовать
│   │   │   └── repairable.go       # всё, что можно чинить
│   │   │
│   │   ├── entities/
│   │   │   ├── car.go
│   │   │   ├── boat.go
│   │   │   ├── plane.go
│   │   │   └── bike.go
│   │   │
│   │   ├── services/
│   │   │   ├── rental.go           # логика аренды (внутри транспорта)
│   │   │   └── maintenance.go      # ремонт и обслуживание
│   │   │
│   │   ├── storage/
│   │   │   └── memory.go           # хранилище состояния транспорта
│   │   │
│   │   └── tariffs/
│   │       └── tariffs.go          # цены на аренду, скидки и коэффициенты
│   │
│   ├── users/                      # всё про клиентов
│   │   ├── entities/
│   │   │   └── client.go           # структура клиента
│   │   │
│   │   ├── services/
│   │   │   ├── auth.go             # регистрация / поиск клиента
│   │   │   └── rental.go           # связывает клиента и транспорт
│   │   │
│   │   └── storage/
│   │       └── memory.go
│   │
│   ├── genid/
│   │   └── genid.go                # генерация уникальных ID (uuid / короткие)
│   │   └── randparameters.go       # генерация рандомных параментров для структур
│   │
│   └── common/
│       ├── errors.go               # общие ошибки, типы, утилиты логирования
│       └── utils.go                # вспомогательные функции общего назначения
│
├── go.mod
└── go.sum


```
